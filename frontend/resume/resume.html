让每次操作增加支持语意: 这次操作的线程及其子线程都已结束的方式

简化版本
void doB(ThreadsafePromiseContainer &promises, size_t index)
{
    cout << "do B" << endl;
    this_thread::sleep_for(seconds(2));

    dispatchThread(promises, doA);
    dispatchThread(promises, doC);

    cout << "set value, index: " << index << endl;
    promises.setValue(index);
}

void dispatchThread(ThreadsafePromiseContainer &promises, function<void(ThreadsafePromiseContainer &, size_t)> fun)
{
    promise<void> aPromise;
    auto size = promises.pushBack(aPromise);
    cout << "size: " << size << endl;
    thread(fun, ref(promises), size - 1).detach();
}

void doBusiness(ThreadsafePromiseContainer &promises)
{
    dispatchThread(promises, doA);
    dispatchThread(promises, doB);
}

int main()
{
    ThreadsafePromiseContainer promises;
    doBusiness(promises);
    promises.waitAllPromisesFinished();

    cout << "all sub task has finished" << endl;
    return 0;
}
适用于corelib代码的优化版本

结合dispatchXXXX中的 complete回调

void dispatchThread(
    function<void(size_t index, function<void(size_t)>)> fun, //fun中去set value
    function<void(size_t)> completion = nullptr,
    ThreadsafePromiseContainer *p = nullptr)
{
    size_t size = 0;
    if (p != nullptr)
    {
        promise<void> aPromise;
        size = p->pushBack(aPromise);
    }
    cout << "size: " << size << endl;
    thread(fun, size - 1, completion).detach();
}

void doBusiness(function<void(size_t)> completion = nullptr,
                ThreadsafePromiseContainer *p = nullptr)
{
    dispatchThread(
        [](size_t index, function<void(size_t)> completion)
        {
            cout << "do something, index:" << index << endl;
            completion(index);
        },
        completion,
        p
    );
}

int main()
{
    ThreadsafePromiseContainer promises;
    auto completion = [&promises](size_t index) {
        cout << "thread<" << index << "> has completed, set value to mark" << endl;
        promises.setValue(index);
    };
    doBusiness(completion, &promises);
    promises.waitAllPromisesFinished();

    cout << "all sub task has finished" << endl;
    return 0;
}
